# Transformation des colonnes Age et EstimatedSalary en types de donnes numriques
df2Num = df2.select(col("User ID"), col("Gender"), col("Age").cast("int"), col("EstimatedSalary").cast("int"), col("Purchased").cast("int"))

# Cration des tranches d'ge personnalises
ageBins = [0, 20, 25, 30, 35, 40, 45, 50, 55, 60, 65, 100]
ageBinLabels = ["0-20", "21-25", "26-30", "31-35", "36-40", "41-45", "46-50", "51-55", "56-60", "61-65", "65+"]

df2Num = df2Num.withColumn("Age Bin", when(col("Age") <= 20, "0-20")
                                           .when(col("Age") <= 25, "21-25")
                                           .when(col("Age") <= 30, "26-30")
                                           .when(col("Age") <= 35, "31-35")
                                           .when(col("Age") <= 40, "36-40")
                                           .when(col("Age") <= 45, "41-45")
                                           .when(col("Age") <= 50, "46-50")
                                           .when(col("Age") <= 55, "51-55")
                                           .when(col("Age") <= 60, "56-60")
                                           .when(col("Age") <= 65, "61-65")
                                           .otherwise("65+"))

# Nombre d'achats par tranche d'ge
agePurchases = df2Num.groupBy("Age Bin").agg(sum("Purchased").alias("Total Purchases")).orderBy("Age Bin").collect()

# Cration du graphique
ageLabels = [row["Age Bin"] for row in agePurchases]
agePurchasesData = [row["Total Purchases"] for row in agePurchases]

fig1, ax1 = plt.subplots()
ax1.bar(ageLabels, agePurchasesData)
plt.title("Nombre d'achats par tranche d'ge")
plt.show()
